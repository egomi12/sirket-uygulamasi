        const getDocRef = (colName, docId) => doc(db, 'artifacts', appId, 'public', 'data', colName, docId);

        // Başlangıç Firebase Kurulumu
        async function initFirebase() {
            try {
                app = initializeApp(firebaseConfig);
                db = getFirestore(app);
                auth = getAuth(app);

                // RULE 3: Önce arka planda yetki al
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }

                // Kullanıcı Giriş Durumu Kontrolü
                checkSession();
            } catch (error) {
                console.error("Firebase başlatılamadı:", error);
                showToast("Sistem bağlantı hatası!", "error");
            }
        }

        // Kullanıcı Oturum Kontrolü
        async function checkSession() {
            const savedUserId = localStorage.getItem('bizim_sirket_user_id');
            if (savedUserId) {
                try {
                    const userDoc = await getDoc(getDocRef('users', savedUserId));
                    if (userDoc.exists()) {
                        currentUser = userDoc.data();
                        setupDashboard();
                        return;
                    }
                } catch (e) {
                    console.error("Kullanıcı verisi alınamadı:", e);
                }
            }
            // Kayıtlı oturum yoksa Giriş ekranını göster
            document.getElementById('auth-container').classList.remove('hidden');
        }

        // KAYIT İŞLEMİ
        window.handleRegister = async function() {
            const name = document.getElementById('reg-name').value.trim();
            const username = document.getElementById('reg-username').value.trim().toLowerCase();
            const title = document.getElementById('reg-title').value.trim();
            const role = document.getElementById('reg-role').value;
            const password = document.getElementById('reg-password').value;

            if (!name || !username || !title || !password) {
                showToast("Lütfen tüm alanları doldurun.", "error");
                return;
            }

            try {
                // RULE 2: No Complex Queries. Tüm kullanıcıları çekip JS tarafında filtreleyeceğiz.
                const querySnapshot = await getDocs(getCol('users'));
                let userExists = false;
                querySnapshot.forEach(doc => {
                    if (doc.data().username === username) {
                        userExists = true;
                    }
                });

                if (userExists) {
                    showToast("Bu kullanıcı adı zaten alınmış.", "error");
                    return;
                }

                const userId = "u_" + Date.now();
                const initials = name.split(' ').map(n => n[0]).join('').toUpperCase().substring(0, 2);
                
                const userData = {
                    id: userId,
                    name: name,
                    username: username,
                    title: title,
                    role: role,
                    password: password, // Gerçek senaryolar için basitleştirilmiş şifreleme/saklama
                    initials: initials,
                    createdAt: new Date().toISOString()
                };

                await setDoc(getDocRef('users', userId), userData);
                
                // Otomatik Giriş
                currentUser = userData;
                localStorage.setItem('bizim_sirket_user_id', userId);
                setupDashboard();
                showToast("Hesabınız oluşturuldu ve giriş yapıldı!", "success");
            } catch (e) {
                showToast("Kayıt sırasında hata oluştu.", "error");
            }
        }

        // GİRİŞ İŞLEMİ
        window.handleLogin = async function() {
            const username = document.getElementById('login-username').value.trim().toLowerCase();
            const password = document.getElementById('login-password').value;

            if (!username || !password) {
                showToast("Lütfen tüm alanları doldurun.", "error");
                return;
            }

            try {
                const querySnapshot = await getDocs(getCol('users'));
                let match = null;
                querySnapshot.forEach(doc => {
                    const u = doc.data();
                    if (u.username === username && u.password === password) {
                        match = u;
                    }
                });

                if (match) {
                    currentUser = match;
                    localStorage.setItem('bizim_sirket_user_id', currentUser.id);
                    setupDashboard();
                    showToast("Başarıyla giriş yapıldı!");
                } else {
                    showToast("Hatalı kullanıcı adı veya şifre.", "error");
                }
            } catch (e) {
                showToast("Giriş yapılırken bir hata oluştu.", "error");
            }
        }

        // ÇIKIŞ İŞLEMİ
        window.handleLogout = async function() {
            if (currentUser) {
                // Presence'tan sil
                await deleteDoc(getDocRef('presence', currentUser.id));
            }
            localStorage.removeItem('bizim_sirket_user_id');
            currentUser = null;
            document.getElementById('auth-container').classList.remove('hidden');
            showToast("Oturum kapatıldı.", "info");
        }

        // ANLIK AKTİFLİK SİSTEMİ (PRESENCE)
        function startPresence() {
            const updatePresence = async () => {
                if (!currentUser) return;
                try {
                    await setDoc(getDocRef('presence', currentUser.id), {
                        userId: currentUser.id,
                        name: currentUser.name,
                        title: currentUser.title,
                        role: currentUser.role,
                        initials: currentUser.initials,
                        lastActive: new Date().toISOString()
                    });
                } catch (e) {
                    console.error("Presence güncellenemedi:", e);
                }
            };

            updatePresence();
            // Her 15 saniyede bir odaya ses ver
            setInterval(updatePresence, 15000);
        }

        // REAL-TIME FİRESTORE DİNLEYİCİLERİ
        function startLiveListeners() {
            // 1. KULLANICI LİSTESİNİ ÇEK VE GÖREV ATAMA KUTUSUNU DOLDUR
            onSnapshot(getCol('users'), (snapshot) => {
                const selectElement = document.getElementById('task-assignee');
                selectElement.innerHTML = '';
                allUsers = {};
                
                snapshot.forEach(doc => {
                    const u = doc.data();
                    allUsers[u.id] = u;
                    
                    const opt = document.createElement('option');
                    opt.value = u.id;
                    opt.text = `${u.name} (${u.title})`;
                    selectElement.appendChild(opt);
                });

                // Ekip sayısını güncelle
                document.getElementById('stat-hr-value').innerText = Object.keys(allUsers).length + " Personel";
                
                // Çalışan Listesini (Sohbet destekli) tekrar çiz
                rebuildTeamChatList();
            }, (err) => console.error("Kullanıcı dinleme hatası:", err));

            // 2. AKTİF ÇALIŞANLARI DİNLE (YEŞİL NOKTA)
            onSnapshot(getCol('presence'), (snapshot) => {
                presenceData = {};
                snapshot.forEach(doc => {
                    const p = doc.data();
                    presenceData[p.userId] = p;
                });
                
                // Çevrimiçi durumlar değiştiğinde de sohbet listesini güncelle
                rebuildTeamChatList();
            }, (err) => console.error("Presence dinleme hatası:", err));

            // 3. GÖREVLERİ DİNLE
            onSnapshot(getCol('tasks'), (snapshot) => {
                const tasksList = [];
                snapshot.forEach(doc => {
                    tasksList.push(doc.data());
                });
                renderTasksUI(tasksList);
            }, (err) => console.error("Görev dinleme hatası:", err));

            // 4. DUYURULARI DİNLE
            onSnapshot(getCol('announcements'), (snapshot) => {
                const annList = [];
                snapshot.forEach(doc => {
                    annList.push(doc.data());
                });
                // En yeni duyuruları üstte göstermek için sırala
                annList.sort((a,b) => b.id - a.id);
                renderAnnouncementsUI(annList);
            }, (err) => console.error("Duyuru dinleme hatası:", err));

            // 5. İZİN TALEPLERİNİ DİNLE
            onSnapshot(getCol('leaves'), (snapshot) => {
                const leavesList = [];
                snapshot.forEach(doc => {
                    leavesList.push(doc.data());
                });
                leavesList.sort((a,b) => b.id - a.id);
                renderLeavesUI(leavesList);
            }, (err) => console.error("İzin dinleme hatası:", err));

            // 6. BELGELERİ DİNLE
            onSnapshot(getCol('documents'), (snapshot) => {
                const docList = [];
                snapshot.forEach(doc => {
                    docList.push(doc.data());
                });
                renderDocumentsUI(docList);
            }, (err) => console.error("Belge dinleme hatası:", err));

            // 7. BİREBİR MESAJLARI DİNLE
            onSnapshot(getCol('direct_messages'), (snapshot) => {
                directMessagesCache = [];
                snapshot.forEach(doc => {
                    directMessagesCache.push(doc.data());
                });
                // En eski mesajlar üstte olsun diye sırala
                directMessagesCache.sort((a,b) => a.timestamp.localeCompare(b.timestamp));
                
                // Eğer bir sohbet aktifse, sohbet penceresini güncelle
                if (activeChatUser) {
                    renderDirectMessagesUI();
                }
            }, (err) => console.error("Mesaj dinleme hatası:", err));
        }

        // TÜM EKİBİ AKTİFLİK DURUMUNA GÖRE SOL MENÜDE LİSTELE VE TIKLAMA DESTEĞİ VER
        function rebuildTeamChatList() {
            const list = document.getElementById('active-employees-list');
            if (!list) return;
            list.innerHTML = '';

            let onlineCount = 0;
            const now = new Date();

            // Tüm kayıtlı kullanıcıları gez
            Object.values(allUsers).forEach(user => {
                // Kendi ismimizi sohbet listesinde göstermeyelim (veya gösterebiliriz de, göstermemek daha iyidir)
                if (currentUser && user.id === currentUser.id) return;

                const presenceInfo = presenceData[user.id];
                let isOnline = false;

                if (presenceInfo) {
                    const lastActiveDate = new Date(presenceInfo.lastActive);
                    const diffSeconds = Math.abs(now - lastActiveDate) / 1000;
                    if (diffSeconds < 45) {
                        isOnline = true;
                        onlineCount++;
                    }
                }

                const row = document.createElement('div');
                row.className = "flex items-center justify-between p-2 rounded-lg hover:bg-gray-100 dark:hover:bg-gray-700/50 transition cursor-pointer group";
                row.onclick = () => openDirectChat(user.id);
                
                const dotClass = isOnline ? "bg-green-500 animate-pulse" : "bg-gray-300 dark:bg-gray-600";
                const dotLabel = isOnline ? "Çevrimiçi" : "Çevrimdışı";

                row.innerHTML = `
                    <div class="flex items-center space-x-2.5">
                        <div class="w-8 h-8 rounded-full bg-gray-100 dark:bg-gray-700 text-gray-800 dark:text-white flex items-center justify-center font-bold text-xs relative">
                            ${user.initials}
                            <span class="absolute -bottom-0.5 -right-0.5 w-2.5 h-2.5 rounded-full border border-white dark:border-gray-800 ${dotClass}"></span>
                        </div>
                        <div>
                            <p class="text-xs font-semibold group-hover:text-primary-600 dark:group-hover:text-primary-400 text-left">${user.name}</p>
                            <p class="text-[9px] text-gray-400 text-left">${user.title}</p>
                        </div>
                    </div>
                    <span class="text-[9px] ${isOnline ? 'text-emerald-500 font-bold' : 'text-gray-400'}">${dotLabel}</span>
                `;
                list.appendChild(row);
            });

            const onlineCounter = document.getElementById('online-counter');
            if (onlineCounter) onlineCounter.innerText = `${onlineCount} Aktif`;
            
            const statOnlineCount = document.getElementById('stat-online-count');
            if (statOnlineCount) statOnlineCount.innerText = `${onlineCount} Aktif`;
        }

            const user = allUsers[userId];
            if (!user) return;

        // CANLI BİREBİR MESAJ GÖNDERME
        window.sendDirectMessage = async function() {
            const input = document.getElementById('chat-message-input');
            const text = input.value.trim();
            if (!text || !activeChatUser || !currentUser) return;

            try {
                const messageId = "msg_" + Date.now();
                const msgData = {
                    id: messageId,
                    senderId: currentUser.id,
                    senderName: currentUser.name,
                    receiverId: activeChatUser.id,
                    receiverName: activeChatUser.name,
                    text: text,
                    timestamp: new Date().toISOString()
                };

                // Firestore'a gönder (RULE 1 ve RULE 3 kurallarına tam uyumlu)
                await setDoc(getDocRef('direct_messages', messageId), msgData);
                
                input.value = '';
                input.focus();
            } catch (e) {
                console.error("Mesaj gönderilemedi:", e);
                showToast("Mesaj iletilemedi.", "error");
            }
        }

        // BİREBİR MESAJLARI ARKA PLANDA SÜZ VE EKRANA ÇİZ
        function renderDirectMessagesUI() {
            const container = document.getElementById('chat-messages-container');
            if (!container || !activeChatUser || !currentUser) return;

            container.innerHTML = '';

            // RULE 2: No Complex Queries. Tüm veriler arasından bellekte filtreleme yapıyoruz:
            const filteredMsgs = directMessagesCache.filter(msg => {
                return (msg.senderId === currentUser.id && msg.receiverId === activeChatUser.id) ||
                       (msg.senderId === activeChatUser.id && msg.receiverId === currentUser.id);
            });

            if (filteredMsgs.length === 0) {
                container.innerHTML = `
                    <div class="text-center py-8 text-gray-400 text-[11px] space-y-1">
                        <i class="fa-regular fa-comments text-xl text-primary-500"></i>
                        <p>Henüz yazışma geçmişiniz bulunmuyor.</p>
                        <p class="text-[9px] text-gray-400/80">İlk mesajı siz atarak işbirliği başlatın!</p>
                    </div>
                `;
                return;
            }

            filteredMsgs.forEach(msg => {
                const isMe = msg.senderId === currentUser.id;
                
                const wrapper = document.createElement('div');
                wrapper.className = `flex ${isMe ? 'justify-end' : 'justify-start'} w-full animate-fade-in`;

                const formattedTime = new Date(msg.timestamp).toLocaleTimeString('tr-TR', { hour: '2-digit', minute: '2-digit' });

                wrapper.innerHTML = `
                    <div class="max-w-[75%] rounded-xl px-3 py-2 text-xs shadow-sm border ${
                        isMe 
                        ? 'bg-primary-600 border-primary-500 text-white rounded-br-none' 
                        : 'bg-white dark:bg-gray-800 border-gray-100 dark:border-gray-700 text-gray-800 dark:text-gray-100 rounded-bl-none'
                    }">
                        <p class="leading-relaxed break-words">${escapeHtml(msg.text)}</p>
                        <span class="block text-[8px] mt-1 text-right ${isMe ? 'text-primary-100' : 'text-gray-400'}">${formattedTime}</span>
                    </div>
                `;
                container.appendChild(wrapper);
            });

            // Otomatik olarak sohbetin en altına kaydır
            container.scrollTop = container.scrollHeight;
        }

        // GÖREV KANBAN ARABİRİMİNİ ÇİZ
        function renderTasksUI(tasks) {
            const colTodo = document.getElementById('col-todo');
            const colDoing = document.getElementById('col-doing');
            const colDone = document.getElementById('col-done');

            colTodo.innerHTML = '';
            colDoing.innerHTML = '';
            colDone.innerHTML = '';

            let todoCount = 0;
            let doingCount = 0;
            let doneCount = 0;

            const isEmployee = currentUser.role === "employee";
            document.getElementById('tasks-header-desc').innerText = isEmployee 
                ? "Bulunduğunuz çalışan yetkisiyle sadece size atanan görevleri görür ve durumlarını değiştirebilirsiniz."
                : "Yönetici / İK yetkisiyle tüm personellerin görevlerini görebilir ve yönlendirebilirsiniz.";

            tasks.forEach(task => {
                // Çalışan filtresi: Çalışanlar sadece kendilerine atanan görevleri görür
                if (isEmployee && task.assigneeId !== currentUser.id) {
                    return;
                }

                const card = document.createElement('div');
                card.className = "bg-white dark:bg-gray-800 p-4 rounded-xl shadow-sm border border-gray-100 dark:border-gray-700 flex flex-col justify-between space-y-3 animate-fade-in";
                
                let catClass = "bg-gray-100 text-gray-600 dark:bg-gray-700 dark:text-gray-300";
                if (task.category === "Tasarım") catClass = "bg-indigo-100 text-indigo-600 dark:bg-indigo-900/30 dark:text-indigo-400";
                if (task.category === "Geliştirme") catClass = "bg-blue-100 text-blue-600 dark:bg-blue-900/30 dark:text-blue-400";
                if (task.category === "İK") catClass = "bg-pink-100 text-pink-600 dark:bg-pink-900/30 dark:text-pink-400";
                if (task.category === "Yönetim") catClass = "bg-emerald-100 text-emerald-600 dark:bg-emerald-900/30 dark:text-emerald-400";

                card.innerHTML = `
                    <div class="space-y-1">
                        <div class="flex justify-between items-center">
                            <span class="text-[9px] font-bold px-2 py-0.5 rounded ${catClass}">${task.category}</span>
                            <span class="text-[9px] text-gray-400 font-bold">${task.assigneeName}</span>
                        </div>
                        <p class="text-xs font-semibold text-gray-800 dark:text-gray-100 leading-snug">${task.title}</p>
                    </div>
                    <div class="flex items-center justify-between border-t pt-2 border-gray-100 dark:border-gray-700">
                        <div class="w-6 h-6 rounded-full bg-primary-600 text-white text-[10px] font-bold flex items-center justify-center" title="Atanan: ${task.assigneeName}">
                            ${task.assigneeInitials || 'P'}
                        </div>
                        <div class="flex gap-1">
                            ${task.status !== 'todo' ? `<button onclick="changeTaskStatus('${task.id}', 'prev')" class="p-1 hover:bg-gray-100 dark:hover:bg-gray-700 rounded text-gray-500"><i class="fa-solid fa-arrow-left text-xs"></i></button>` : ''}
                            ${currentUser.role === 'admin' ? `<button onclick="deleteTask('${task.id}')" class="p-1 hover:bg-red-50 dark:hover:bg-red-950/30 rounded text-red-500"><i class="fa-regular fa-trash-can text-xs"></i></button>` : ''}
                            ${task.status !== 'done' ? `<button onclick="changeTaskStatus('${task.id}', 'next')" class="p-1 hover:bg-gray-100 dark:hover:bg-gray-700 rounded text-gray-500"><i class="fa-solid fa-arrow-right text-xs"></i></button>` : ''}
                        </div>
                    </div>
                `;

                if (task.status === 'todo') {
                    colTodo.appendChild(card);
                    todoCount++;
                } else if (task.status === 'doing') {
                    colDoing.appendChild(card);
                    doingCount++;
                } else if (task.status === 'done') {
                    colDone.appendChild(card);
                    doneCount++;
                }
            });

        // KANBAN SÜRÜKLEME / DURUM DEĞİŞTİRME
        window.changeTaskStatus = async function(taskId, direction) {
            try {
                const taskRef = getDocRef('tasks', taskId);
                const taskSnap = await getDoc(taskRef);
                if (!taskSnap.exists()) return;

                const task = taskSnap.data();
                const stages = ['todo', 'doing', 'done'];
                let currentIndex = stages.indexOf(task.status);

                if (direction === 'next' && currentIndex < 2) {
                    await updateDoc(taskRef, { status: stages[currentIndex + 1] });
                } else if (direction === 'prev' && currentIndex > 0) {
                    await updateDoc(taskRef, { status: stages[currentIndex - 1] });
                }
                showToast("Görev durumu güncellendi.");
            } catch (e) {
                showToast("Görev durumu güncellenemedi.", "error");
            }
        }

                    <i class="fa-solid fa-spinner animate-spin text-purple-600"></i>
                    <p class="text-xs text-gray-500">Gemini yanıtı formüle ediyor...</p>
                </div>
            `;
            chatArea.appendChild(loadingMsg);
            chatArea.scrollTop = chatArea.scrollHeight;

            let success = false;
            let retryCount = 0;
            const maxRetries = 5;
            let delay = 1000;
            let aiResponseText = "";

            const payload = {

            };

            while (retryCount < maxRetries && !success) {
                try {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=`, {
                        method: 'POST',

                } catch (error) {
                    retryCount++;
                    if (retryCount >= maxRetries) {
                        aiResponseText = "Şu anda yapay zeka sunucusunda bir yoğunluk var. Lütfen biraz sonra tekrar deneyin.";
                    } else {
                        await new Promise(resolve => setTimeout(resolve, delay));
                        delay *= 2;
                    }
                }
            }

            const loadingElement = document.getElementById('ai-loading');
            if (loadingElement) loadingElement.remove();

            const aiMsg = document.createElement('div');
            aiMsg.className = "flex gap-3 mb-4 animate-fade-in";
            aiMsg.innerHTML = `
                <div class="w-8 h-8 rounded-full bg-purple-600 text-white flex items-center justify-center font-bold text-xs shrink-0">
                    AI
                </div>
                <div class="bg-white dark:bg-gray-800 p-3 rounded-xl shadow-sm border border-gray-100 dark:border-gray-700 max-w-2xl">
                    <div class="text-sm whitespace-pre-wrap text-gray-800 dark:text-gray-100 leading-relaxed">${escapeHtml(aiResponseText)}</div>
                </div>
            `;
            chatArea.appendChild(aiMsg);
            chatArea.scrollTop = chatArea.scrollHeight;
        }

        function escapeHtml(text) {
            return text
                .replace(/&/g, "&amp;")
                .replace(/</g, "&lt;")
                .replace(/>/g, "&gt;")
                .replace(/"/g, "&quot;")
                .replace(/'/g, "&#039;");
        }

        // Temayı ilk yüklemede ayarla
        if (localStorage.getItem('theme') === 'dark' || (!('theme' in localStorage) && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
            document.documentElement.classList.add('dark');
        } else {
            document.documentElement.classList.remove('dark');
        }
